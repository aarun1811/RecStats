    import { useEffect } from 'react'

    import { DashboardRenderer } from '@/components/dashboard/dashboard-renderer'
    import { EmbedTopbar } from '@/components/embed/embed-topbar'
    import { useTheme } from '@/components/layout/theme-provider'
    import { Skeleton } from '@/components/ui/skeleton'
    import { useManagedDashboard } from '@/hooks/use-managed-dashboards'
    import {
      parseFilterParams,
      parseLockedFilters,
      parseHideTokens,
    } from '@/lib/dashboard-url-state'

    export const Route = createFileRoute('/embed/dashboards/$dashboardId')({
      component: EmbedDashboardPage,
      validateSearch: (search: Record<string, unknown>) => search,
    })

    function EmbedDashboardPage() {
      const { dashboardId } = Route.useParams()
      const search = Route.useSearch()
      const { data: dashboard, isLoading } = useManagedDashboard(dashboardId)
      const config = dashboard?.config

      const initialFilters = parseFilterParams(search)
      const lockedFilters = parseLockedFilters(search)
      const hideTokens = parseHideTokens(search)

      // Build filterParams query string for the "Open in RecViz" link in EmbedTopbar.
      // Preserve existing behavior: include all filter.* keys (including filter.lock) so the
      // recipient sees the same state in the regular view route.
      const filterParamsEntries: string[] = []
      for (const [key, val] of Object.entries(search)) {
        if (key.startsWith('filter.') && typeof val === 'string') {
          filterParamsEntries.push(
            `${encodeURIComponent(key)}=${encodeURIComponent(val)}`,
          )
        }
      }
      const filterParams = filterParamsEntries.join('&')

      // Apply theme from URL via ThemeProvider (preserved pre-existing behavior)
      const { setTheme: applyTheme } = useTheme()
      const themeParam = typeof search.theme === 'string' ? search.theme : undefined
      useEffect(() => {
        if (themeParam === 'dark' || themeParam === 'light') {
          applyTheme(themeParam)
        }
      }, [themeParam, applyTheme])

      if (isLoading) {
        return (
          <div className="h-screen flex flex-col">
            <div className="h-9 border-b bg-muted/30" />
            <div className="p-6 space-y-4 flex-1">
              <Skeleton className="h-12 w-full" />
              <div className="grid grid-cols-4 gap-4">
                {Array.from({ length: 4 }).map((_, i) => (
                  <Skeleton key={i} className="h-24" />
                ))}
              </div>
            </div>
          </div>
        )
      }

      if (!dashboard || !config) {
        return <div className="p-6">Dashboard not found</div>
      }

      return (
        <div className="h-screen flex flex-col">
          <EmbedTopbar
            title={dashboard.name}
            dashboardId={dashboardId}
            filterParams={filterParams}
            hideTitle={hideTokens.has('title')}
          />
          <div className="p-6 flex-1 overflow-auto space-y-6">
            <DashboardRenderer
              config={config}
              initialFilters={initialFilters}
              lockedFilters={lockedFilters}
              hideFilterBar={hideTokens.has('filter-bar')}
              hideToolbar={hideTokens.has('toolbar')}
            />
          </div>
        </div>
      )
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Host portal ↔ iframe embed URL | Host portal controls the full URL including `?hide=` tokens; tokens are untrusted input |
| Browser URL ↔ embed route | Search params flow through `parseFilterParams` / `parseHideTokens` before reaching React state |
| Embed iframe ↔ backend | Same boundary as any dashboard load — managed_dashboards endpoint, Pydantic validated |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-9-1 | Tampering / Info Disclosure | URL filter params rendered in DOM via embed route | mitigate | React 19 auto-escapes JSX; filter values pass through store as data. Acceptance criterion in Tasks 2 + 3 + 4 asserts zero new `dangerouslySetInnerHTML`. |
| T-9-4 | Tampering | `?hide=<script>` or other malicious hide token | mitigate | `parseHideTokens` returns `Set<string>`. Each rendering branch checks `hide.has('its-token')` — string equality, no execution. Unknown tokens are no-op. |
| T-9-6 | Information Disclosure | Embed route loads any dashboard without auth | accept | D-11 defers auth to v2. Corporate intranet trust assumption documented. Future SSO phase will add `frame-ancestors` header + auth gate. |
| T-9-12 | Clickjacking | Iframe embed clickjacked by external site | accept | D-11 trust assumption. Future SSO phase should add `Content-Security-Policy: frame-ancestors` response header. Not in Phase 9 scope. |
| T-9-13 | Tampering | `filter.lock` bypass by omitting the lock param | accept | Lock is a UX constraint, not a security control. The backend still applies RLS when it arrives in the SSO phase. Document as "lock is UX-only in v1". |
