    ```python
    """Fernet symmetric encryption for database credentials at rest."""

    from __future__ import annotations

    from cryptography.fernet import Fernet, InvalidToken


    class EncryptionService:
        """Encrypt/decrypt database passwords using Fernet (AES-128-CBC + HMAC-SHA256).

        Key must be a URL-safe base64-encoded 32-byte key (generated by Fernet.generate_key()).
        """

        def __init__(self, key: str) -> None:
            self._fernet = Fernet(key.encode() if isinstance(key, str) else key)

        def encrypt(self, plaintext: str) -> str:
            """Encrypt a plaintext password. Returns URL-safe base64 ciphertext."""
            return self._fernet.encrypt(plaintext.encode()).decode()

        def decrypt(self, ciphertext: str) -> str:
            """Decrypt a ciphertext back to plaintext password."""
            return self._fernet.decrypt(ciphertext.encode()).decode()

        @staticmethod
        def generate_key() -> str:
            """Generate a new Fernet key suitable for RECVIZ_ENCRYPTION_KEY."""
            return Fernet.generate_key().decode()
    ```

    ```python
    def build_async_uri(
        backend: str,
        host: str,
        port: int | None = None,
        database: str | None = None,
        username: str | None = None,
        password: str | None = None,
    ) -> str:
        """Build an async-compatible SQLAlchemy URI for create_async_engine.

        PostgreSQL: postgresql+asyncpg://user:pass@host:port/dbname
        Oracle: oracle+oracledb://user:pass@host:port/?service_name=SID
        """
        dialect_prefix = ASYNC_DIALECTS.get(backend)
        if dialect_prefix is None:
            raise ValueError(f"Unsupported async backend: {backend}. Supported: {list(ASYNC_DIALECTS.keys())}")

        port = port or DEFAULT_PORTS.get(backend, 5432)
        user_part = ""
        if username:
            encoded_pass = quote_plus(password) if password else ""
            user_part = f"{quote_plus(username)}:{encoded_pass}@"

        if backend == "oracle":
            service_name = database or "ORCL"
            return f"{dialect_prefix}://{user_part}{host}:{port}/?service_name={service_name}"

        if backend == "postgresql":
            db_name = database or "postgres"
            return f"{dialect_prefix}://{user_part}{host}:{port}/{db_name}"

        # Should not reach here due to ASYNC_DIALECTS check above
        raise ValueError(f"Unsupported backend: {backend}")
    ```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-02 | Information Disclosure | encrypted_password column | mitigate | Fernet encryption at rest via EncryptionService. Password never stored in plaintext. Ciphertext is URL-safe base64 with HMAC authentication. |

| T-12-04 | Information Disclosure | URI with password in logs | mitigate | URIs built at runtime from decrypted fields, passed directly to create_async_engine. SQLAlchemy masks passwords in engine.url repr. Never log the raw URI string. |
| T-12-05 | Tampering | Encrypted password integrity | accept | Fernet includes HMAC-SHA256 authentication -- tampered ciphertext raises InvalidToken on decrypt. No additional integrity check needed. |
| T-12-06 | Spoofing | EncryptionService with weak key | accept | Fernet.generate_key() produces cryptographically random 32-byte keys. Dev default key is acceptable for local PostgreSQL with no real credentials. |
