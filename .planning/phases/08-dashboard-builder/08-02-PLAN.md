   import type { ChartLayout, FilterConfig } from '@/types/dashboard-config'

   export type BuilderItemType = 'chart' | 'kpi' | 'grid'

   export interface BuilderChartRef {
     chartId: string        // Reference to RecvizChart.id from library
     chartType: string      // e.g. 'bar', 'line', etc.
     datasetId: string      // For filter column matching
     title: string          // Display title (can be overridden)
     crossFilter: boolean   // Whether this chart participates in cross-filtering
     drillHierarchy: string[] // Drill-down column path
     drillDetailDataSourceId: string | null
     refreshInterval: number | null // Per-chart refresh override (ms)
   }

   export interface BuilderKpiRef {
     kpiId: string          // Reference to RecvizKpi.id from library
     title: string          // Display title
   }

   export interface BuilderGridRef {
     datasetId: string      // Dataset for the grid
     title: string          // Display title
     visibleColumns: string[] | null // null = show all
     defaultSortColumn: string | null
     defaultSortDirection: 'asc' | 'desc'
     rowLimit: number       // 50 | 100 | 500 | 1000
   }

   export interface BuilderItem {
     id: string             // Unique panel id (crypto.randomUUID)
     type: BuilderItemType
     layout: ChartLayout    // { col, row, width, height }
     chart?: BuilderChartRef
     kpi?: BuilderKpiRef
     grid?: BuilderGridRef
   }

   export interface BuilderState {
     dashboardId: string | null  // null for new dashboards
     name: string
     description: string
     items: BuilderItem[]
     filters: FilterConfig[]
     isDirty: boolean
   }

2. Create `frontend/src/stores/builder-store.ts` (Zustand store):
   - State: `dashboardId: string | null`, `name: string`, `description: string`, `items: BuilderItem[]`, `filters: FilterConfig[]`, `isDirty: boolean`
   - Actions:
     - `initNew()` — reset store for a new dashboard (empty state, isDirty=false)
     - `initFromConfig(id: string, config: DashboardConfig)` — populate store from a loaded config, isDirty=false
     - `updateName(name: string)` — set name, isDirty=true
     - `updateDescription(desc: string)` — set description, isDirty=true
     - `addItem(item: BuilderItem)` — append to items, isDirty=true
     - `removeItem(itemId: string)` — filter out item, isDirty=true
     - `updateLayouts(layouts: Array<{id: string; layout: ChartLayout}>)` — update layout for each item, isDirty=true
     - `updateItemConfig(itemId: string, updates: Partial<BuilderChartRef> | Partial<BuilderKpiRef> | Partial<BuilderGridRef>)` — update the type-specific ref on an item, isDirty=true
     - `addFilter(filter: FilterConfig)` — append to filters, isDirty=true
     - `removeFilter(filterId: string)` — filter out, isDirty=true
     - `updateFilter(filterId: string, updates: Partial<FilterConfig>)` — merge updates, isDirty=true
     - `reorderFilters(orderedIds: string[])` — reorder filters array by given id order, isDirty=true
     - `markClean()` — set isDirty=false (called after save)
   - `initFromConfig` must parse the DashboardConfig and create BuilderItem[] from `config.charts` and `config.grids`, mapping ChartLayout fields. For charts, create BuilderChartRef from DashboardChartConfig. For grids, create BuilderGridRef from GridConfig. Filters are copied directly.

3. Create `frontend/src/stores/layout-history-store.ts` (Zustand store per D-07):
   - State: `past: ChartLayout[][]`, `future: ChartLayout[][]` (each entry is a snapshot of all item layouts)
   - Computed: `canUndo` (past.length > 0), `canRedo` (future.length > 0)
   - Actions:
     - `pushSnapshot(layouts: ChartLayout[])` — push current to past, clear future, cap past at 50 entries
     - `undo()` — pop from past, push to future, return the popped layout
     - `redo()` — pop from future, push to past, return the popped layout
     - `reset()` — clear past and future (called on Save per D-07)
   - MAX_HISTORY constant = 50
