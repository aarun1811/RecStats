2. Add config serialization function to `builder-page.tsx`:

Create a `serializeConfig` function that converts builder store state to a valid `DashboardConfig`:

```typescript
function serializeConfig(state: BuilderStoreState): DashboardConfig {
  const charts: DashboardChartConfig[] = state.items
    .filter(item => item.type === 'chart' && item.chart)
    .map(item => ({
      id: item.id,
      title: item.chart!.title,
      type: item.chart!.chartType,
      sourceType: 'query' as const,
      sources: [{ dataSourceId: item.chart!.datasetId, metric: '' }], // dataset is the source
      layout: item.layout,
      crossFilter: item.chart!.crossFilter,
      drillHierarchy: item.chart!.drillHierarchy.length > 0 ? item.chart!.drillHierarchy : undefined,
      drillDetailDataSourceId: item.chart!.drillDetailDataSourceId ?? undefined,
      refreshInterval: item.chart!.refreshInterval ?? undefined,
    }))

  const grids: GridConfig[] = state.items
    .filter(item => item.type === 'grid' && item.grid)
    .map(item => ({
      id: item.id,
      title: item.grid!.title,
      dataSourceId: item.grid!.datasetId,
      columns: [], // populated by the renderer from the dataset
      layout: item.layout,
    }))

  return {
    id: state.dashboardId ?? crypto.randomUUID(),
    name: state.name,
    description: state.description,
    features: { crossFilter: true, drillDown: true },
    filters: state.filters,
    kpis: [], // KPI items are rendered as panels, not in the old KpiConfig[] format
    charts,
    grids,
    layout: { type: 'custom', sections: ['filters', 'charts', 'grids'] },
    autoRefreshInterval: 600000,
  }
}
```

1. Create `frontend/src/components/builder/unsaved-changes-guard.tsx` (D-29):

```typescript
interface UnsavedChangesGuardProps {
  isDirty: boolean
  onConfirmLeave: () => void
  onCancelLeave: () => void
  open: boolean
}
```

Two-part guard:
a) **Browser tab close**: `useEffect` that adds `beforeunload` event when `isDirty`:
   ```typescript
   useEffect(() => {
     if (!isDirty) return
     const handler = (e: BeforeUnloadEvent) => { e.preventDefault() }
     window.addEventListener('beforeunload', handler)
     return () => window.removeEventListener('beforeunload', handler)
   }, [isDirty])
   ```

b) **In-app navigation dialog**:
   ```tsx
   <Dialog open={open} onOpenChange={(o) => { if (!o) onCancelLeave() }}>
     <DialogContent className="max-w-md">
       <DialogHeader>
         <DialogTitle>Unsaved changes</DialogTitle>
         <DialogDescription>You have unsaved changes. Leave without saving?</DialogDescription>
       </DialogHeader>
       <DialogFooter className="gap-2 sm:gap-0">
         <Button variant="outline" onClick={onCancelLeave}>Keep Editing</Button>
         <Button variant="destructive" onClick={onConfirmLeave}>Leave Without Saving</Button>
       </DialogFooter>
     </DialogContent>
   </Dialog>
   ```
